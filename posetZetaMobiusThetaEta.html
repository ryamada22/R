my.zeta <- function(g,rho){
  D_rho <- distances(g,v=rho)
  A_poset <- A
  for(i in 1:nv){
    for(j in 1:nv){
      if(D_rho[i] >= D_rho[j]){
        A_poset[i,j] <- 0
      }
    }
  }
  g_poset <- graph.adjacency(A_poset)
  zeta <- distances(g_poset,mode="out")
  zeta[which(is.infinite(zeta))] <- 0
  zeta <- matrix(zeta>0,nv,nv) + 0
  diag(zeta) <- 1
  return(zeta)
}
